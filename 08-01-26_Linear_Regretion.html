x = df["Height (in cm)"].values
y = df["Weight (in kg)"].values

x_mean = np.mean(x)
y_mean = np.mean(y)
In[18]:
numerator = np.sum((x - x_mean) * (y - y_mean))
denominator = np.sum((x - x_mean) ** 2)

m = numerator / denominator
c = y_mean - m * x_mean

m, c
